//Menu-driven program for database operations(create,insert,update,delete)
import java.sql.*;
import java.util.*;
class DatabaseAllOperations
{
	public static void main(String args[])throws Exception 
	{
		Class.forName("oracle.jdbc.driver.OracleDriver");
		System.out.println("Driver Register Successfully!!!");
		Connection con=DriverManager.getConnection("jdbc:oracle:thin:@localhost:1521:XE","system","system");
		System.out.println("Connection Established Successfully!!!");
		Scanner sc=new Scanner(System.in);
		int id,price,ch;
		do
		{
			System.out.println("****DATABASE OPERATIONS****");
			System.out.println("1. Create Table");
			System.out.println("2. Insert row");
			System.out.println("3. Update row");
			System.out.println("4. Delete row");
			System.out.println("5. Fetch Records");
			System.out.println("6. Exit");
			System.out.println("Enter Your Choice:");
			ch=sc.nextInt();
			switch(ch)
			{
				case 1: Statement stmt=con.createStatement();
						stmt.execute("create table Book(bookid number(4),bookname varchar(10),bookprice number(3))");
						System.out.println("Table created successfully");
						break;
				case 2: PreparedStatement pstmt=con.prepareStatement("insert into Book values(?,?,?)");
						System.out.println("Enter Book ID:");
						id=sc.nextInt();
						System.out.println("Enter Book Name:");
						String nm=sc.next();
						System.out.println("Enter Book Price:");
						price=sc.nextInt();
						pstmt.setInt(1,id);
						pstmt.setString(2,nm);
						pstmt.setInt(3,price);
						pstmt.executeUpdate();
						System.out.println("Row inserted successfully!!!");
						break;
				case 3: PreparedStatement pstmt1=con.prepareStatement("update Book set bookprice=? where bookid=?");
						System.out.println("Enter Book ID for Updation:");
						id=sc.nextInt();
						System.out.println("Enter New Book Price:");
						price=sc.nextInt();
						pstmt1.setInt(1,price);
						pstmt1.setInt(2,id);
						pstmt1.executeUpdate();
						System.out.println("Row Updated Successfully!!!");
						break;
				case 4: PreparedStatement pstmt2=con.prepareStatement("delete from Book where bookid=?");
						System.out.println("Enter Book ID for Deletion:");
						id=sc.nextInt();
						pstmt2.setInt(1,id);
						pstmt2.executeUpdate();
						System.out.println("Row Deleted Successfully!!!");
						break;
				case 5: Statement stmt1=con.createStatement();
						ResultSet rs=stmt1.executeQuery("select * from Book");
						System.out.println("*******BOOK MNGT SYSTEM*******");
						System.out.println("BOOKID\tNAME\tPRICE");
						System.out.println("\n============================");
						while(rs.next())
						{
							System.out.println(rs.getInt(1)+"\t"+rs.getString(2)+"\t"+rs.getInt(3));
						}
						break;
				case 6: System.out.println("Thanks for using our software");
						break;
				default:System.out.println("Invalid Choice!!!");
			}
		}while(ch!=6);
		con.close();
	}
}
